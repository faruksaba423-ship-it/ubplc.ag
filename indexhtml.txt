        <button type="submit" id="submitBtn">প্রোফাইল সেভ করুন</button>
    </form>
    <p id="status" style="text-align:center; color: green;"></p>

    <div id="memberList">
        <h3>সকল সদস্য তালিকা</h3>
        <button onclick="loadMembers()" style="background: #6c757d; margin-bottom: 15px;">তালিকা রিফ্রেশ করুন</button>
        <div id="listContent">লোড হচ্ছে...</div>
    </div>

<script>
    const scriptURL = 'https://script.google.com/macros/s/AKfycbwFWz8GSzGeA-tGwOcr8DHbIyU4AjsCMsG9_oypMhQ8/exec';
    
    // তথ্য সেভ করার ফাংশন
    document.getElementById('ubplcForm').addEventListener('submit', e => {
        e.preventDefault();
        const btn = document.getElementById('submitBtn');
        btn.disabled = true;
        fetch(scriptURL, { method: 'POST', body: new FormData(e.target)})
        .then(() => {
            document.getElementById('status').innerText = "সফলভাবে সেভ হয়েছে!";
            e.target.reset();
            btn.disabled = false;
            loadMembers(); // সেভ হওয়ার পর অটো লিস্ট আপডেট হবে
        });
    });

    // শিট থেকে তথ্য এনে দেখানোর ফাংশন (৩০০ জনের জন্য)
    function loadMembers() {
        const listDiv = document.getElementById('listContent');
        listDiv.innerHTML = "লোড হচ্ছে...";
        
        // আপনার শিট থেকে ডেটা পড়ার জন্য নিচের লিঙ্কটি ব্যবহার হবে (JSON ফরম্যাটে)
        // এটি একটু অ্যাডভান্সড, তাই আপাতত আমরা সিম্পল রাখছি। 
        // ৩০০ জন হলে আপনি সরাসরি Google Sheet এর CSV লিঙ্ক ব্যবহার করতে পারেন।
        fetch(scriptURL + "?read=true") 
        .then(res => res.json())
        .then(data => {
            listDiv.innerHTML = "";
            data.forEach(member => {
                listDiv.innerHTML += `
                    <div class="member-card">
                        <h4>${member.name}</h4>
                        <p>পদবী: ${member.designation}</p>
                        <p>ফোন: ${member.phone}</p>
                    </div>`;
            });
        }).catch(() => {
            listDiv.innerHTML = "আপাতত তালিকা দেখার জন্য শিটটি চেক করুন।";
        });
    }
</script>
